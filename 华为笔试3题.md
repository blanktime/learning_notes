# 从数组中选择n个数，使得和为m，给出方案种数
---

```c++
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

int count(int i)
{
    int c = 0;
    while (i)
    {
        if (i & 1)
        {
            ++c;
        }
        i >>= 1;
    }
    return c;
}

int Select(vector<int> arr, int number, int sum)
{
    int len = arr.size();
    int res = 0;
    for (int i = 0; i < pow(2, len); i++)
    {
        if (count(i) == number)
        {
            int s = 0;
            for (int j = 0; j < len; j++)
            {
                if (i & 1 << (len - j - 1))
                {
                    s += arr[j];
                }
            }
            if (s == sum)
            {
                res++;
            }
        }
    }
    return res;
}

int main()
{
    int m, value;
    cin >> m >> value;
    int num;
    cin >> num;
    vector<int> v;
    while (num--)
    {
        int temp;
        cin >> temp;
        v.push_back(temp);
    }
    //sort(v.begin(), v.end());
    int res = Select(v, m, value);
    cout << res;
    return 0;
}
```
